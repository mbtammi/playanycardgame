// Firebase Firestore Security Rules for Newsletter App (Free Plan Compatible)
// Copy these rules to your Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Newsletter emails collection - write-only for public, read for admin
    match /newsletter_emails/{document=**} {
      allow create: if true; // Anyone can add emails
      allow read, update, delete: if false; // Only admin through backend
    }
    
    // Deny all other collections
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
